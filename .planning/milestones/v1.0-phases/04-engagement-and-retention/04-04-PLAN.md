    Create `generateSearchName` utility per RESEARCH.md example:
    ```typescript
    export function generateSearchName(filters: {
      category?: string | null;
      countyName?: string | null;
      query?: string | null;
    }): string {
      const parts: string[] = [];
      if (filters.category) parts.push(filters.category);
      if (filters.query) parts.push(`"${filters.query}"`);
      if (!parts.length) parts.push('All listings');
      if (filters.countyName) parts.push(`in ${filters.countyName}`);
      return parts.join(' ');
    }
    ```

    **POST /api/saved-searches (+server.ts):**
    Follow established API pattern (CSRF check, auth check):
    - Parse JSON body: `{ category?: string, county?: string, locality?: string, query?: string, name?: string }`
    - IMPORTANT: `county` must store the county ID (e.g., `ie/leinster/dublin`), NOT the display name. The cron worker matches against `location_profile_data->'county'->>'id'`.
    - If no `name` provided, auto-generate using `generateSearchName()`. To get the county display name for the auto-name, import `getCountyOptionById` from `$lib/location-hierarchy` and look up the county ID.
    - Insert into `saved_searches` table: `{ user_id: user.id, category, county, locality, query, name, notify: true }` (notify defaults to true for new saved searches)
    - Return `json({ success: true, id: inserted.id })`

    **DELETE /api/saved-searches/[id] (+server.ts):**
    - CSRF check, auth check
    - Delete WHERE `id = params.id AND user_id = user.id`
    - Return `json({ success: true })`

    ```typescript
    import type { PageServerLoad } from './$types';

    export const load: PageServerLoad = async ({ locals }) => {
      const { data: { user } } = await locals.supabase.auth.getUser();
      if (!user) return { searches: [] };

      const { data, error } = await locals.supabase
        .from('saved_searches')
        .select('id, name, category, county, locality, query, notify, created_at')
        .eq('user_id', user.id)
        .order('created_at', { ascending: false });

      return { searches: data ?? [] };
    };
    ```
