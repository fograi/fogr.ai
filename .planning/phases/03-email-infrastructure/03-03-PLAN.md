```typescript
async function sendApprovalEmail(env: Env, ad: PendingAd): Promise<void> {
	const emailEnv = buildEmailEnv(env);
	if (!env.RESEND_API_KEY) return; // Silently skip if email not configured

	// Check preferences -- ad_approved is suppressible
	const suppressed = await isEmailSuppressed(emailEnv, ad.user_id, 'ad_approved');
	if (suppressed) return;

	const userEmail = await getUserEmail(env, ad.user_id);
	if (!userEmail) return;

	const adUrl = `https://fogr.ai/ad/${ad.slug ?? ad.id}`;
	const token = await generateUnsubscribeToken(
		env.UNSUBSCRIBE_SECRET ?? '',
		ad.user_id,
		'ad_approved'
	);
	const unsubUrl = `https://fogr.ai/api/unsubscribe?token=${encodeURIComponent(token)}&type=ad_approved`;

	const html = renderEmail(
		'Your listing is live on fogr.ai',
		buildAdApprovedEmailHtml({ adTitle: ad.title, adUrl, unsubscribeUrl: unsubUrl })
	);

	await sendEmail(emailEnv, {
		to: userEmail,
		subject: 'Your listing is live on fogr.ai',
		html,
		headers: buildUnsubscribeHeaders(unsubUrl)
	});
}
```

```typescript
async function sendRejectionEmail(env: Env, ad: PendingAd): Promise<void> {
	const emailEnv = buildEmailEnv(env);
	if (!env.RESEND_API_KEY) return;

	// Rejection is a moderation/DSA email -- do NOT check preferences, do NOT add unsubscribe headers
	const userEmail = await getUserEmail(env, ad.user_id);
	if (!userEmail) return;

	const html = renderEmail(
		'Your fogr.ai listing was not approved',
		buildAdRejectedEmailHtml({
			adTitle: ad.title,
			adId: ad.id,
			reason: 'Content does not meet community guidelines.'
		})
	);

	await sendEmail(emailEnv, {
		to: userEmail,
		subject: 'Your fogr.ai listing was not approved',
		html
		// No headers -- no List-Unsubscribe on DSA/moderation emails
	});
}
```

**Determine the recipient:**
After the message is successfully inserted (after the `if (msgErr)` check and after the conversation update), determine who should receive the notification:

- The recipient is the OTHER user in the conversation -- NOT the sender
- If `user.id === conversation.seller_id` -> recipient is `conversation.buyer_id` (but we might not have conversation in scope if it was just looked up by adId)
- More robustly: the recipient is the person who is not `user.id`. If `user.id === ad.user_id` (seller is sending), recipient is the buyer. If `user.id !== ad.user_id` (buyer is sending), recipient is `ad.user_id` (the seller).
- Use the conversation object if available: `const recipientId = user.id === (conversation?.seller_id ?? ad.user_id) ? (conversation?.buyer_id ?? user.id) : (conversation?.seller_id ?? ad.user_id);`
- Simplest: `const recipientId = isSeller ? (conversation?.buyer_id ?? '') : ad.user_id;`
  - Note: if `isSeller` is true, we must have a conversation (sellers can only reply in existing conversations). So `conversation?.buyer_id` is always available when isSeller is true.
  - If buyer is messaging: recipient is always `ad.user_id` (the seller).

**Send notification (fire-and-forget, after the message insert succeeds):**
Do this AFTER the existing `recordMetric` calls, right before the final `return json(...)`:

```typescript
// Fire-and-forget email notification to recipient
// Wrapped in an async IIFE that we do NOT await on the response path
// to avoid slowing down the message API response.
// However, in Cloudflare Workers, we should use waitUntil if available
// to ensure the email send completes after the response is sent.
const emailPromise = (async () => {
	try {
		const env: EmailEnv = {
			RESEND_API_KEY: platform?.env?.RESEND_API_KEY ?? '',
			PUBLIC_SUPABASE_URL: platform?.env?.PUBLIC_SUPABASE_URL ?? '',
			SUPABASE_SERVICE_ROLE_KEY: platform?.env?.SUPABASE_SERVICE_ROLE_KEY ?? '',
			UNSUBSCRIBE_SECRET: platform?.env?.UNSUBSCRIBE_SECRET ?? ''
		};
		if (!env.RESEND_API_KEY) return; // Email not configured

		const recipientId = isSeller ? (conversation?.buyer_id ?? '') : ad.user_id;
		if (!recipientId) return;

		const suppressed = await isEmailSuppressed(env, recipientId, 'messages');
		if (suppressed) return;

		// Look up recipient email via Supabase auth admin API
		const authUrl = new URL(`/auth/v1/admin/users/${recipientId}`, env.PUBLIC_SUPABASE_URL);
		const authRes = await fetch(authUrl, {
			headers: {
				apikey: env.SUPABASE_SERVICE_ROLE_KEY,
				Authorization: `Bearer ${env.SUPABASE_SERVICE_ROLE_KEY}`
			}
		});
		if (!authRes.ok) return;
		const authUser = (await authRes.json()) as { email?: string };
		if (!authUser.email) return;

		// Build ad URL using slug if available (fetch slug from ad row)
		const adSlugQuery = await locals.supabase
			.from('ads')
			.select('slug')
			.eq('id', ad.id)
			.maybeSingle();
		const adSlug = adSlugQuery.data?.slug ?? ad.id;
		const adUrl = `https://fogr.ai/ad/${adSlug}`;

		const token = await generateUnsubscribeToken(env.UNSUBSCRIBE_SECRET, recipientId, 'messages');
		const unsubUrl = `https://fogr.ai/api/unsubscribe?token=${encodeURIComponent(token)}&type=messages`;

		const html = renderEmail(
			'You have a new message on fogr.ai',
			buildNewMessageEmailHtml({
				adTitle: ad.title ?? 'your listing',
				adUrl,
				unsubscribeUrl: unsubUrl
			})
		);

		await sendEmail(env, {
			to: authUser.email,
			subject: 'You have a new message on fogr.ai',
			html,
			headers: buildUnsubscribeHeaders(unsubUrl)
		});
	} catch (err) {
		console.error('message_notification_email_failed', { error: String(err) });
	}
})();

// Use waitUntil if available to ensure email completes after response
if (platform?.context?.waitUntil) {
	platform.context.waitUntil(emailPromise);
}
```
