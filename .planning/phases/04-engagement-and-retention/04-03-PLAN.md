  <action>
    Create `/api/watchlist` endpoint following the established pattern from `src/routes/api/ads/[id]/status/+server.ts`:

    **POST** (save ad to watchlist):
    - CSRF check via `isSameOrigin(request, url)`
    - Auth check via `locals.supabase.auth.getUser()` -- return 401 if not logged in
    - Parse JSON body: `{ ad_id: string }`
    - Validate ad_id is non-empty
    - Insert into `watchlist` table: `{ user_id: user.id, ad_id }`
    - Handle unique constraint violation gracefully (upsert or catch 23505 error code) -- if already saved, return 200 success (idempotent)
    - Return `json({ success: true })`

    **DELETE** (unsave ad from watchlist):
    - CSRF check and auth check (same as POST)
    - Parse JSON body: `{ ad_id: string }`
    - Delete from `watchlist` where `user_id = user.id AND ad_id = ad_id`
    - Return `json({ success: true })` (idempotent -- deleting non-existent is fine)

    **GET** (check if ad is saved -- for ad page to know initial state):
    - Auth check -- return `{ saved: false }` if not logged in (no error, just false)
    - Read `ad_id` from URL search params: `url.searchParams.get('ad_id')`
    - Query watchlist for matching row
    - Return `json({ saved: !!data })`

    Import `isSameOrigin` from `$lib/server/csrf`. Use `json` from `@sveltejs/kit`.

    **Watchlist page server loader (src/routes/(app)/watchlist/+page.server.ts):**
    Follow the pattern from `src/routes/(app)/ads/+page.server.ts`:
    ```typescript
    import type { PageServerLoad } from './$types';
    import type { AdCard } from '../../../types/ad-types';

    export const load: PageServerLoad = async ({ locals }) => {
      const { data: { user } } = await locals.supabase.auth.getUser();
      if (!user) return { ads: [] as AdCard[] };

      const { data, error } = await locals.supabase
        .from('watchlist')
        .select('ad_id, created_at, ads(id, slug, title, description, price, currency, category, category_profile_data, location_profile_data, image_keys, created_at, status, firm_price, min_offer, sale_price)')
        .eq('user_id', user.id)
        .order('created_at', { ascending: false });

      // Map joined ads data to AdCard format
      // Filter out any watchlist entries where the ad was deleted
      const ads: AdCard[] = (data ?? [])
        .filter(item => item.ads)
        .map(item => {
          const ad = item.ads as any;
          return {
            id: ad.id,
            slug: ad.slug ?? undefined,
            title: ad.title,
            price: ad.price ?? null,
            img: ad.image_keys?.[0] ?? '',
            description: ad.description ?? '',
            category: ad.category ?? '',
            categoryProfileData: ad.category_profile_data ?? null,
            locationProfileData: ad.location_profile_data ?? null,
            currency: ad.currency ?? undefined,
            firmPrice: ad.firm_price ?? false,
            minOffer: ad.min_offer ?? null,
            createdAt: ad.created_at ?? undefined,
            status: ad.status ?? undefined,
            salePrice: ad.sale_price ?? null
          };
        });

      return { ads };
    };
    ```

      import AdCard from '$lib/components/AdCard.svelte';
      import type { AdCard as AdCardType } from '../../../types/ad-types';

      export let data: { ads: AdCardType[] };
      let ads = data.ads;
      let removingId: string | null = null;

      async function unsave(adId: string) {
        removingId = adId;
        try {
          const res = await fetch('/api/watchlist', {
            method: 'DELETE',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify({ ad_id: adId })
          });
          if (res.ok) ads = ads.filter(a => a.id !== adId);
        } finally {
          removingId = null;
        }
      }
    </script>
